583 
584  if (new_location == Location::host) {
585  device_id = 0;
586  }
587 
588  if (new_location == _location) {
589  return;
590  }
591 
592 #ifndef LINALG_NO_CHECKS
593  else if ((new_location != Location::host) && (_location != Location::host)){
594 
595  throw excUnimplemented("Dense.location(): Moves only to and from main "
596  "memory supported. Try moving to main memory first "
597  "and move to target from there.");
598  // The reason for this is that it is tedious to guarantee a consistent state
599  // in all cases. It is easier if the user handles potential failures.
600  }
601 #endif
602 
603  // The matrix is empty, we just update the meta data
604  if (_rows == 0) {
605 
606  _location = new_location;
607  _device_id = device_id;
608 
609  return;
610 
611  };

614  // don't deallocate yet)
615  Dense<T> tmp;
616  tmp.clone_from(*this);
617 
618  // Reallocate the memory on the target
619  this->reallocate(_rows, _cols, new_location, device_id);
620 
621  // Copy the data over
