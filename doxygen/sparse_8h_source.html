548 void Sparse<T>::location(Location new_location, int device_id) {
549 
550  if (new_location == Location::host) {
551  device_id = 0;
552  }
553 
554  if (new_location == _location) {
555  return;
556  }
557 #ifndef LINALG_NO_CHECKS
558  else if ((new_location != Location::host) && (_location != Location::host)) {
559 
560  throw excUnimplemented("Moves only to and from main memory supported. "
561  "Try moving to main memory first and move to "
562  "target from there.");
563  // The reason for this is that it is tedious to guarantee a consistent state
564  // in all cases. It is easier if the user handles potential failures.
565  }
566 #endif
567 #ifdef HAVE_CUDA
568  else if ((new_location == Location::GPU) && (_location == Location::host)) {
569 
570 #ifdef WARN_COPY
571  std::cout << "Warning: copying matrix to GPU\n";
572 #endif
573 
574  using CUDA::cuda_make_shared;
575  using Utilities::copy_1Darray;
576 
577  int prev_device;
578  checkCUDA(cudaGetDevice(&prev_device));
579 
580  // Allocate memory on GPU
581  checkCUDA(cudaSetDevice(device_id));
582  auto device_values = cuda_make_shared<T>(_n_nonzeros, _device_id);
583  auto device_indices = cuda_make_shared<I_t>(_n_nonzeros, _device_id);
584  auto device_edges = cuda_make_shared<I_t>(_size + 1, _device_id);
585 
586  // Copy matrix to GPU
587  copy_1Darray(_values.get(), _n_nonzeros, device_values.get(), _location,
588  _device_id, new_location, device_id);
589  copy_1Darray(_indices.get(), _n_nonzeros, device_indices.get(), _location,
590  _device_id, new_location, device_id);
591  copy_1Darray(_edges.get(), _size + 1, device_edges.get(), _location,
592  _device_id, new_location, device_id);
593 
594  checkCUDA(cudaSetDevice(prev_device));
595 
596  // Change members to reflect the change
597  _location = new_location;
598  _device_id = device_id;
599 
600  // This frees the memory on the host
601  _values = device_values;
602  _indices = device_indices;
603  _edges = device_edges;
604 
605 
606  }
607  else if ((new_location == Location::host) && (_location == Location::GPU)) {
608

610  std::cout << "Warning: copying matrix from GPU\n";
611 #endif
612 
613  using Utilities::host_make_shared;
614  using Utilities::copy_1Darray;
615

621  auto host_indices = host_make_shared<I_t>(_n_nonzeros);
622  auto host_edges = host_make_shared<I_t>(_size + 1);
623 
624  // Copy matrix to GPU
625  checkCUDA(cudaSetDevice(_device_id));
626  copy_1Darray(_values.get(), _n_nonzeros, host_values.get(), _location,
627  _device_id, new_location, device_id);
628  copy_1Darray(_indices.get(), _n_nonzeros, host_indices.get(), _location,
629  _device_id, new_location, device_id);
630  copy_1Darray(_edges.get(), _size + 1, host_edges.get(), _location,
631  _device_id, new_location, device_id);
632 
633  checkCUDA(cudaSetDevice(prev_device));
634 
635  // Change members to reflect the change
636  _location = new_location;
637  _device_id = device_id;
638 
639  // This frees the memory on the device
640  _values = host_values;
641  _indices = host_indices;
642  _edges = host_edges;
643 
644  }
645 #endif
