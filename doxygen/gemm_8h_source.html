231  check_output_transposed(C, "xGEMM()");

233  if (A.rows() != C.rows() || A.cols() != B.rows() || B.cols() != C.cols()) {
234  throw excBadArgument("xGEMM(): argument matrix size mismatch");
235  }
236 #endif

238  auto location = A._location;
239  auto device_id = A._device_id;
240  auto m = A.rows();
241  auto n = B.cols();
242  auto k = B.rows();
243  auto A_ptr = A._begin();
244  auto lda = A._leading_dimension;
245  auto B_ptr = B._begin();
246  auto ldb = B._leading_dimension;
247  auto C_ptr = C._begin();
248  auto ldc = C._leading_dimension;
249 
250  if (location == Location::host) {
251  char transa = (A._transposed) ? 'T' : 'N';
252  char transb = (B._transposed) ? 'T' : 'N';
253  xGEMM(transa, transb, m, n, k, alpha, A_ptr, lda, B_ptr, ldb, beta, C_ptr,
254  ldc);
255  }

259  cublasOperation_t transb = (B._transposed) ? CUBLAS_OP_T : CUBLAS_OP_N;
260  BLAS::CUBLAS::xGEMM(handles[device_id], transa, transb, m, n, k, &alpha,
261  A_ptr, lda, B_ptr, ldb, &beta, C_ptr, ldc);
