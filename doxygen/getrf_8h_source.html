228 #ifndef HAVE_CUDA
229  if (A.rows() != ipiv.rows()) {
230  throw excBadArgument("xGETRF(): argument matrix size mismatch");

232 #else
233  if (A.rows() != A.cols() && A._location == Location::GPU) {
234  throw excBadArgument("xGETRF(): matrix A must be a square matrix (CUBLAS "
